# PUTTY
add-apt-repository ppa:certbot/certbot
apt-get update
apt-get install python-certbot-apache
certbot --apache -d www.tortuga-finance.com
-- https://www.ssllabs.com/ssltest/analyze.html?d=www.tortuga-finance.com&latest
certbot renew --dry-run  -- test if everything works fine

# EXAMPLE CONF FILE
WSGIDaemonProcess financeprocess python-path=/home/finance_project/finance python-home=/home/finance_project/financeenv
<VirtualHost *:80>
        ServerName www.tortuga-finance.com
        ServerAdmin daniel@tortuga-webdesign.de
        # redirect
        RewriteEngine On
        RewriteCond "%{HTTP_HOST}" "!^www\.tortuga-finance\.com$" [OR]
        RewriteCond "%{SERVER_PORT}" "!^443$"
        RewriteRule "^" "https://www.tortuga-finance.com/%{REQUEST_URI}"
        # logs
        ErrorLog ${APACHE_LOG_DIR}/financeerror.log
        CustomLog ${APACHE_LOG_DIR}/financeaccess.log combined
        # process
	WSGIProcessGroup financeprocess
        WSGIScriptAlias / /home/finance_project/finance/config/wsgi.py
        # alias
        Alias /static /home/finance_project/finance/finance/static
        Alias /media /home/finance_project/finance/finance/media
        # dirs
        <Directory /home/finance_project/finance/finance/static>
                Require all granted
        </Directory>
        <Directory /home/finance_project/finance/finance/media>
                Require all granted
        </Directory>
        <Directory /home/finance_project/finance/config>
                <Files wsgi.py>
                        Require all granted
                </Files>
        </Directory>
</VirtualHost>


# EXAMPLE SSL CERTBOT CONF FILE
<IfModule mod_ssl.c>
<VirtualHost *:443>
        ServerName www.tortuga-finance.com
        ServerAdmin daniel@tortuga-webdesign.de
        # redirect
        RewriteEngine On
        RewriteCond "%{HTTP_HOST}" "!^www\.tortuga-finance\.com$" [OR]
        RewriteCond "%{SERVER_PORT}" "!^443$"
        RewriteRule "^" "https://www.tortuga-finance.com/%{REQUEST_URI}"
        # logs
        ErrorLog ${APACHE_LOG_DIR}/financeerror.log
        CustomLog ${APACHE_LOG_DIR}/financeaccess.log combined
        # process
	WSGIProcessGroup financeprocess
        WSGIScriptAlias / /home/finance_project/finance/config/wsgi.py
        # alias
        Alias /static /home/finance_project/finance/finance/static
        Alias /media /home/finance_project/finance/finance/media
        # dirs
        <Directory /home/finance_project/finance/finance/static>
                Require all granted
        </Directory>
        <Directory /home/finance_project/finance/finance/media>
                Require all granted
        </Directory>
        <Directory /home/finance_project/finance/config>
                <Files wsgi.py>
                        Require all granted
                </Files>
        </Directory>
        # certbot
        SSLCertificateFile /etc/letsencrypt/live/www.tortuga-finance.com/fullchain.pem
        SSLCertificateKeyFile /etc/letsencrypt/live/www.tortuga-finance.com/privkey.pem
        Include /etc/letsencrypt/options-ssl-apache.conf
</VirtualHost>
</IfModule>
